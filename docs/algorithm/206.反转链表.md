class Solution
{
public:
    ListNode *reverseList(ListNode *head)
    {
        ListNode *temp = nullptr;
        ListNode *fast = head;
        ListNode *slow = nullptr;

        while (fast)
        {
            temp = fast->next;
            fast->next = slow;
            slow = fast;
            fast = temp;
        }
        return slow;
    }
};
