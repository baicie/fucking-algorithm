# 非原始值的响应式方案

## 理解 Proxy 和 Reflect

Proxy 它能够实现对其他对象的代理，这里的关键词是其他对象  
代理: 指的是对一个对象基本语义的代理。它允许我们拦截并重新定义对一个对象的基本操作

***在副作用函数内通过原始对象访问它的某个属性是不会建立响应联系的***

代理对象的 get 拦截函数接收第三个参数 receiver，它代表谁在读取属性

```js
        const bucket = new WeakMap()
        let activeEffect
        const effectStack = []

        const obj = {
            foo: 1, get bar() {
                return this.foo
            }
        }
        const p = new Proxy(obj, {
            get(target, key, receiver) {
                track(target, key)
                // this由原始对象obj变成了代理对象p
                return Reflect.get(target, key, receiver)
            },
            set(target, key, newVal) {
                target[key] = newVal
                trigger(target, key)
            }
        })

        function effect(fn, options = {}) {
            const effectFn = () => {
                cleanup(effectFn)
                activeEffect = effectFn
                effectStack.push(effectFn)
                const res = fn()
                effectStack.pop()
                activeEffect = effectStack[effectStack.length - 1]
                return res
            }
            effectFn.options = options
            // 这些deps和effectFn存在关联关系
            effectFn.deps = []
            if (!options.lazy) {
                effectFn()
            }
            return effectFn
        }

        function cleanup(effectFn) {
            for (let i = 0; i < effectFn.deps.length; i++) {
                const deps = effectFn.deps[i]
                //删除所有存在关系的deps中的当前的effectFn
                deps.delete(effectFn)
            }
            effectFn.deps.length = 0
        }

        function track(target, key) {
            if (!activeEffect) return
            // target映射
            let depsMap = bucket.get(target)
            if (!depsMap) {
                bucket.set(target, (depsMap = new Map()))
            }
            // key - effect映射
            let deps = depsMap.get(key)
            if (!deps) {
                depsMap.set(key, (deps = new Set()))
            }
            deps.add(activeEffect)
            activeEffect.deps.push(deps)
        }

        function trigger(target, key) {
            const depsMap = bucket.get(target)
            if (!depsMap) return
            const effects = depsMap.get(key)

            const effectsToRun = new Set()
            effects && effects.forEach(effectFn => {
                if (activeEffect !== effectFn) {
                    effectsToRun.add(effectFn)
                }
            })
            effectsToRun.forEach(effectFn => {
                if (effectFn.options.scheduler) {
                    effectFn.options.scheduler(effectFn)
                } else {
                    effectFn()
                }
            })
        }

        effect(() => {
            console.log(p.bar);
        })
        p.foo++
```

```js
                return this.foo
```

```js
        const p = reactive(obj)
        function reactive(obj) {
            return new Proxy(obj, {
                get(target, key, receiver) {
                    if (key === 'raw') {
```

## 只读和浅只读

- 只读不用收集依赖关系
- 只读不用触发

```js
        const bucket = new WeakMap()
        let activeEffect
        const ITERATE_KEY = Symbol()
        const effectStack = []

        function createReactive(obj, isShallow = false, isReadonly = false) {
            return new Proxy(obj, {
                get(target, key, receiver) {
                    if (key === 'raw') {
                        return target
                    }
                    if (!isReadonly) {
                        track(target, key)
                    }
                    const res = Reflect.get(target, key, receiver)

                    if (isShallow) {
                        return res
                    }
                    if (typeof res === "object" && res !== null) {
                        return isReadonly ? readonly(res) : reactive(res)
                    }
                    return res
                },
                set(target, key, newVal, receiver) {
                    if (isReadonly) {
                        console.warn(`${key} is read only`);
                        return true
```
